## 根据一个bool属性将元素分组

* 根据对象数组中，对象的属性的`true`or`false`来二次划分数组

```js
// 输入
const users = [
  { user: 'barney', age: 36, active: false },
  { user: 'fred', age: 40, active: true },
];
// 输出
[
  [{ user: 'fred', age: 40, active: true }],
  [{ user: 'barney', age: 36, active: false }]
]

// by 余光
function groupBy(array, pro) {
    if (!Array.isArray(array)) return 'TypeError';
    const trueA = [];
    const failA = [];
    array.forEach(v => {
        if (v[pro]) {
            trueA.push(v)
        } else {
            failA.push(v)
        }
    });
    array = [
        trueA,
        failA
    ];
    return array;
}
```
